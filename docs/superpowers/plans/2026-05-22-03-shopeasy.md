# ShopEasy MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the ShopEasy product — a Nigerian e-commerce storefront where users browse products, get AI-powered persona-aware recommendations, manage a cart, and view order history.

**Architecture:** Same OOP pattern as InsideNaija — thin routers → Service → Repository (BaseRepository). Frontend features live in `frontend_v2/src/features/shopeasy/`. Persona is built from onboarding answers and persisted in Supabase; recommendations hit the existing FastAPI recommend agent.

**Tech Stack:** FastAPI, supabase-py, BaseRepository, React, TanStack Query, Zustand, shadcn/ui

**Prerequisite:** Foundation plan must be complete. Runs in parallel with InsideNaija plan (no shared code changes).

```sql
-- Products (seeded from existing catalog)
create table products (
  id uuid primary key default gen_random_uuid(),
  external_id text unique,  -- maps to existing product IDs in the catalog JSON
  name text not null,
  description text,
  category text not null default 'general',
  price_naira numeric(12, 2),
  image_url text,
  seller text,
  in_stock boolean not null default true,
  created_at timestamptz not null default now()
);

-- User personas (one per user, upserted)
create table personas (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null unique references users(id) on delete cascade,
  register_tier text not null default 'nigerian_english',
  hedonic_utilitarian numeric(3,2) not null default 0.5,
  communal_individual numeric(3,2) not null default 0.5,
  aspect_priority jsonb not null default '{}',
  language_preference text not null default 'english',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

-- Cart items
create table cart_items (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  product_id uuid not null references products(id) on delete cascade,
  quantity integer not null default 1 check (quantity > 0),
  created_at timestamptz not null default now(),
  unique (user_id, product_id)
);

-- Orders
create table orders (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  status text not null default 'placed',  -- placed | confirmed | shipped | delivered
  total_naira numeric(12,2) not null,
  created_at timestamptz not null default now()
);

-- Order items (snapshot of product at order time)
create table order_items (
  id uuid primary key default gen_random_uuid(),
  order_id uuid not null references orders(id) on delete cascade,
  product_id uuid not null references products(id),
  product_name text not null,
  quantity integer not null,
  unit_price_naira numeric(12,2) not null
);

-- RLS
alter table products enable row level security;
alter table personas enable row level security;
alter table cart_items enable row level security;
alter table orders enable row level security;
alter table order_items enable row level security;

-- Products: anyone authenticated can read
create policy "products: authenticated read" on products
  for select using (auth.uid() is not null);

-- Personas: own only
create policy "personas: own only" on personas
  for all using (auth.uid() = user_id);

-- Cart: own only
create policy "cart_items: own only" on cart_items
  for all using (auth.uid() = user_id);

-- Orders: own only
create policy "orders: own only" on orders
  for all using (auth.uid() = user_id);

create policy "order_items: own order only" on order_items
  for all using (
    order_id in (select id from orders where user_id = auth.uid())
  );
```

from __future__ import annotations
from typing import Any, Optional
from app.db.base_repository import BaseRepository


class ProductRepository(BaseRepository):
    table_name = "products"

    def search(self, query: str, category: Optional[str] = None,
               limit: int = 50) -> list[dict[str, Any]]:
        q = (
            self.client.table(self.table_name)
            .select("*")
            .ilike("name", f"%{query}%")
            .eq("in_stock", True)
        )
        if category:
            q = q.eq("category", category)
        res = q.limit(limit).execute()
        return res.data or []

    def list_by_category(self, category: Optional[str] = None,
                         limit: int = 50) -> list[dict[str, Any]]:
        q = self.client.table(self.table_name).select("*").eq("in_stock", True)
        if category:
            q = q.eq("category", category)
        res = q.limit(limit).execute()
        return res.data or []
```

class PersonaRepository(BaseRepository):
    table_name = "personas"

    def find_by_user(self, user_id: str) -> Optional[dict[str, Any]]:
        res = (
            self.client.table(self.table_name)
            .select("*")
            .eq("user_id", user_id)
            .maybe_single()
            .execute()
        )
        return res.data

    def upsert(self, user_id: str, data: dict[str, Any]) -> dict[str, Any]:
        import datetime
        res = (
            self.client.table(self.table_name)
            .upsert({**data, "user_id": user_id,
                     "updated_at": datetime.datetime.utcnow().isoformat()},
                    on_conflict="user_id")
            .execute()
        )
        return res.data[0]
```

from __future__ import annotations
from typing import Any
from app.db.base_repository import BaseRepository


class CartRepository(BaseRepository):
    table_name = "cart_items"

    def find_by_user(self, user_id: str) -> list[dict[str, Any]]:
        res = (
            self.client.table(self.table_name)
            .select("*, products(*)")
            .eq("user_id", user_id)
            .execute()
        )
        return res.data or []

    def upsert_item(self, user_id: str, product_id: str, quantity: int) -> dict[str, Any]:
        res = (
            self.client.table(self.table_name)
            .upsert({"user_id": user_id, "product_id": product_id, "quantity": quantity},
                    on_conflict="user_id,product_id")
            .execute()
        )
        return res.data[0]

    def remove_item(self, user_id: str, product_id: str) -> None:
        self.client.table(self.table_name).delete() \
            .eq("user_id", user_id).eq("product_id", product_id).execute()

    def clear(self, user_id: str) -> None:
        self.client.table(self.table_name).delete().eq("user_id", user_id).execute()
```

from __future__ import annotations
from typing import Any, Optional
from app.db.repositories.product_repository import ProductRepository
from app.db.repositories.persona_repository import PersonaRepository


class ProductService:
    def __init__(self) -> None:
        self.repo = ProductRepository()
        self.persona_repo = PersonaRepository()

    def list(self, category: Optional[str] = None, search: Optional[str] = None,
             limit: int = 50) -> list[dict[str, Any]]:
        if search:
            return self.repo.search(search, category=category, limit=limit)
        return self.repo.list_by_category(category=category, limit=limit)

    def get(self, product_id: str) -> Optional[dict[str, Any]]:
        return self.repo.find(product_id)

    async def get_recommendations(self, product_id: str,
                                   user_id: str) -> list[dict[str, Any]]:
        from app.agents.recommend_agent import recommend
        from app.api.schemas.persona import Persona

        persona_data = self.persona_repo.find_by_user(user_id)
        persona = Persona(**persona_data) if persona_data else None

        results, _ = await recommend(
            persona=persona,
            candidate_set=None,
            top_k=6,
        )
        product_ids = [r.get("product_id") or r.get("id") for r in results]
        return [self.repo.find(pid) for pid in product_ids if pid]
```

from __future__ import annotations
from typing import Any
from app.db.repositories.order_repository import OrderRepository
from app.db.repositories.cart_repository import CartRepository


class OrderService:
    def __init__(self) -> None:
        self.order_repo = OrderRepository()
        self.cart_repo = CartRepository()

    def place_order(self, user_id: str) -> dict[str, Any]:
        cart = self.cart_repo.find_by_user(user_id)
        if not cart:
            raise ValueError("Cart is empty")

        items = [
            {
                "product_id": item["product_id"],
                "product_name": item["products"]["name"],
                "quantity": item["quantity"],
                "unit_price_naira": float(item["products"].get("price_naira") or 0),
            }
            for item in cart
        ]
        total = sum(i["quantity"] * i["unit_price_naira"] for i in items)
        order = self.order_repo.create_with_items(user_id, total, items)
        self.cart_repo.clear(user_id)
        return order

    def list_orders(self, user_id: str) -> list[dict[str, Any]]:
        return self.order_repo.find_by_user(user_id)
```

import { supabase } from '@/lib/supabase'
import type { Product, CartItem, Order } from './types'

const BASE = import.meta.env.VITE_API_BASE_URL

async function authHeaders(): Promise<HeadersInit> {
  const { data } = await supabase.auth.getSession()
  return {
    'Content-Type': 'application/json',
    Authorization: `Bearer ${data.session?.access_token ?? ''}`,
  }
}

export async function listProducts(params?: { category?: string; search?: string }): Promise<Product[]> {
  const q = new URLSearchParams()
  if (params?.category) q.set('category', params.category)
  if (params?.search) q.set('search', params.search)
  const res = await fetch(`${BASE}/shop/products?${q}`, { headers: await authHeaders() })
  if (!res.ok) throw new Error('Failed to fetch products')
  return res.json()
}

export async function getProduct(id: string): Promise<Product> {
  const res = await fetch(`${BASE}/shop/products/${id}`, { headers: await authHeaders() })
  if (!res.ok) throw new Error('Product not found')
  return res.json()
}

export async function getRecommendations(productId: string): Promise<Product[]> {
  const res = await fetch(`${BASE}/shop/products/${productId}/recommendations`, { headers: await authHeaders() })
  if (!res.ok) return []
  return res.json()
}

export async function getCart(): Promise<CartItem[]> {
  const res = await fetch(`${BASE}/shop/cart`, { headers: await authHeaders() })
  if (!res.ok) throw new Error('Failed to fetch cart')
  return res.json()
}

export async function addToCart(productId: string, quantity = 1): Promise<CartItem> {
  const res = await fetch(`${BASE}/shop/cart`, {
    method: 'POST',
    headers: await authHeaders(),
    body: JSON.stringify({ product_id: productId, quantity }),
  })
  if (!res.ok) throw new Error('Failed to add to cart')
  return res.json()
}

export async function removeFromCart(productId: string): Promise<void> {
  await fetch(`${BASE}/shop/cart/${productId}`, {
    method: 'DELETE',
    headers: await authHeaders(),
  })
}

export async function placeOrder(): Promise<Order> {
  const res = await fetch(`${BASE}/shop/orders`, {
    method: 'POST',
    headers: await authHeaders(),
  })
  if (!res.ok) throw new Error('Failed to place order')
  return res.json()
}

export async function listOrders(): Promise<Order[]> {
  const res = await fetch(`${BASE}/shop/orders`, { headers: await authHeaders() })
  if (!res.ok) throw new Error('Failed to fetch orders')
  return res.json()
}

import { create } from 'zustand'
import type { CartItem } from '@/features/shopeasy/types'
import { addToCart, removeFromCart, getCart } from '@/features/shopeasy/api'

interface CartState {
  items: CartItem[]
  loading: boolean
  fetchCart: () => Promise<void>
  add: (productId: string, quantity?: number) => Promise<void>
  remove: (productId: string) => Promise<void>
}

export const useCartStore = create<CartState>((set, get) => ({
  items: [],
  loading: false,

  fetchCart: async () => {
    set({ loading: true })
    const items = await getCart()
    set({ items, loading: false })
  },

  add: async (productId, quantity = 1) => {
    await addToCart(productId, quantity)
    await get().fetchCart()
  },

  remove: async (productId) => {
    await removeFromCart(productId)
    set((s) => ({ items: s.items.filter((i) => i.product_id !== productId) }))
  },
}))
```

import { useEffect } from 'react'
import { useNavigate } from 'react-router-dom'
import { useMutation, useQueryClient } from '@tanstack/react-query'
import { useCartStore } from '@/store/cart'
import { placeOrder } from '@/features/shopeasy/api'
import { Button } from '@/components/ui/button'

export default function Cart() {
  const navigate = useNavigate()
  const { items, fetchCart, remove } = useCartStore()
  const queryClient = useQueryClient()

  useEffect(() => { fetchCart() }, [fetchCart])

  const orderMutation = useMutation({
    mutationFn: placeOrder,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['orders'] })
      navigate('/shop/orders')
    },
  })

  const total = items.reduce((sum, item) =>
    sum + item.quantity * (item.products?.price_naira ?? 0), 0)

  return (
    <div className="min-h-screen bg-slate-50">
      <header className="bg-white border-b px-6 py-4 flex items-center gap-4">
        <button onClick={() => navigate('/shop')} className="text-slate-400 hover:text-slate-700">←</button>
        <h1 className="text-xl font-bold">Your Cart</h1>
      </header>

      <main className="max-w-xl mx-auto px-6 py-8 space-y-4">
        {items.length === 0 && (
          <div className="text-center py-16 space-y-2">
            <p className="text-slate-500">Your cart is empty</p>
            <Button variant="outline" onClick={() => navigate('/shop')}>Browse products</Button>
          </div>
        )}

        {items.map((item) => (
          <div key={item.id} className="bg-white rounded-xl border p-4 flex items-center gap-4">
            <div className="flex-1">
              <p className="text-sm font-semibold">{item.products?.name}</p>
              <p className="text-xs text-slate-400">Qty: {item.quantity}</p>
            </div>
            <p className="text-sm font-bold">

            </p>
            <button className="text-slate-300 hover:text-red-500 text-sm"
              onClick={() => remove(item.product_id)}>✕</button>
